- lombok 설정 : STS 설정 및 설치 이후 처음 프로젝트 생성 시 딱 한 번만 실행하면 된다.(다음 프로젝트부터는 설정 필요X)
	1. Project and External Dependencies → lombok-x.xx.xx.jar 파일이 존재하는지 확인
						ex)lombok-1.18.46.jar
	2. lombok-x.xx.xx.jar 에서 마우스 우클릭 → Run as → 2.JavaApplication 클릭
	3. 팝업창이 뜰 경우 확인
	4. Specify location 클릭 
	   → 스프링 설치 경로의 SpringToolSuited.exe 파일 위치 선택 후 Select
	   → 경로 맞으면 Install/Update 클릭 
	   → Install successful 이 확인되면 설치 성공 
	   → Quit Installer 클릭하면 설치 완료. 	5. STS 재실행하지 않으면 에러 발생할 수 있음. 반드시 재실행!!

- 탭에서 Spring Boot 띄우기
   · 정적인 웹 리소스
	→ Spring Boot는 프로젝트가 생성될 때 Web 라이브러리가 포함된 경우
	   정적인 웹 리소스는
	   src > main > resources < static 아래에서 관리하도록 설정된다
	   즉, html, css, js, image 등의 정적 콘텐츠는 
	   static 아래에 위치시키면 접근 가능하며, static이 루트 폴더가 된다. 
	→ 프로젝트 생성하여 라이브러리를 추가할 때 Web라이브러리를 추가했다면
	   static 폴더가 자동으로 생성되는데, 선택하지 않았다 하더라도 
	   build.gradle 파일에 Dependency를 추가하고 static 폴더를 직접 만들어주면 된다. 

dependencies {

build.gradle====================================================================

plugins {
	// 1. 이 프로젝트가 자바 기반임을 선언하고 컴파일, 빌드 가능 부여.
	id 'java'

	// 2. 결과물을 JAR가 아니라 WAR 파일로 만든다는 설정 → JSP 사용시 필수  ★★부트는 jsp랑 친화적이지 않다 그래서 실무는jar를많이?
	id 'war'

	// 3. 스프링 부트 플로그인 → 프로젝트의 중심 엔진 역할 수행.
	id 'org.springframework.boot' version '3.5.14'

	// 4. 스프링 부트가 관리하는 라이브러리들의 버전을 자동으로 맞춰주는 헬퍼 설정.
	id 'io.spring.dependency-management' version '1.1.7'
}

// 5. 프로젝트를 만든 그룹 지정. → 보통 회사 도메인의 역순 
group = 'com.doit'

// 6. 현재 이 프로젝트(프로그램)의 버전 → SNAPSHOT 은 개발중이라는 의미.
version = '0.0.1-SNAPSHOT'

// 7. 프로젝트에 대한 짧은 설명.
description = 'Setting Project for Spring Boot'

java {

	toolchain {

		// 8. 이 프로젝트를 컴파일하고 실행할 JDK 버전을 21로 설정. 
		languageVersion = JavaLanguageVersion.of(21)
	}
}

repositories {

	// 9. 아래(→ dependencies) 라이브러리들을 다운로드할 리포지토리.
	mavenCentral()
}

// check!! ★build.gradle에서 젤 중요!!
dependencies {

	// 10. 스프링에서 DB 연결 및 트랜잭션 관리를 처리해 주는 기본 라이브러리.
	implementation 'org.springframework.boot:spring-boot-starter-jdbc'

	// 11. MVC 패턴, RESTful API, 내장 톰캣을 포함한 웹 개발 필수 라이브러리 스타터.
	implementation 'org.springframework.boot:spring-boot-starter-web'

	// 12. MyBatis를 스프링 부트에서 사용할 수 있게 해주는 외부 라이브러리 → 3.0.5 버전 지정.
	implementation 'org.mybatis.spring.boot:mybatis-spring-boot-starter:3.0.5'

	// 13. 롬복 코드 슬림화 지원 라이브러리 
	compileOnly 'org.projectlombok:lombok'

	// 14. 개발 편의 도구 → 코드를 수정하고 저장화면 서버를 알아서 재시작해주는 개발용 도구.
	developmentOnly 'org.springframework.boot:spring-boot-devtools'

	// 15. DB 드라이버 및 서버 환경 설정 → Oracle에 접속하기 위한 JDBC 드라이버로 실행될 때만 필요하므로 runtimeonly사용
	runtimeOnly 'com.oracle.database.jdbc:ojdbc11'

	// 16. 롬복의 어노테이션을 해석해서 실제 자바 코드를 바꿔주는 프로세서.
	annotationProcessor 'org.projectlombok:lombok'

	// 17. 2번의 WAR 플러그인과 짝궁. 외장 톰캣에 배포할 때 내장 톰캣이 충돌 발생하지 않도록 처리. → 충돌 방지.
	providedRuntime 'org.springframework.boot:spring-boot-starter-tomcat'
	
	// 18. 테스트 환경 라이브러리 → JUnit, Mockito 등 자바 코드를 테스트하기 위한 통합 테스트 라이브러리.
	testImplementation 'org.springframework.boot:spring-boot-starter-test'

	// 19. 테스트 환경 라이브러리 → MyBatis 매퍼 파일들이 DB와 잘 연동되는지 테스트할 수 있게 해주는 라이브러리
	testImplementation 'org.mybatis.spring.boot:mybatis-spring-boot-starter-test:3.0.5'
	
	// 20. 테스트 환경 라이브러리 → 테스트 코드를 짤 때도 롬복을 인식할 수 있도록 처리
	testCompileOnly 'org.projectlombok:lombok'

	// 21. 테스트 환경 라이브러리 → jUnit 기반으로 테스트를 원할하게 구동하기 위한 런쳐.
	testRuntimeOnly 'org.junit.platform:junit-platform-launcher'

	// 22. 테스트 환경 라이브러리 → 테스트 코드를 짤 때도 롬복 어노테이션을 인식할 수 있도록 처리.
	testAnnotationProcessor 'org.projectlombok:lombok'
	

	// check!! 직접 추가한 내용!!!
	// tomcat-embed
	// → 23. JSP 사용을 위한 핵심 엔진
	// → 스프링 부트는 기본적으로 JSP에 친화적이지 않다. 
	// → JSP 파일을 서블릿 자바 코드로 변환을 수행하는 핵심 엔진
	implementation 'org.apache.tomcat.embed:tomcat-embed-jasper'
	
	//JSTL
	// implementation 'javax.servlet:jstl' // 스프링부터 3.0 미만
	implementation 'jakarta.servlet:jakarta.servlet-api'
	// → 24. 스프링 부트 3.x 이후부터는 javax가 아닌 jakarts 서블릿 규격을 사용하므로 이를 명시적으로 처리. 

	implementation 'jakarta.servlet.jsp.jstl:jakarta.servlet.jsp.jstl-api'
	// → 25. JSP 화면에서 <c:if>, <c:forEach> 등을 사용하기 위한 API 규격

	implementation 'org.glassfish.web:jakarta.servlet.jsp.jstl'
	// → 26. 25 규격을 실제 구현해 놓은 외부 라이브러리 → 25, 26 둘 다 있어야 JSTL 이 돌아간다.
	
}

tasks.named('test') {

	// → 27. 우리가 작성한 테스트 코드를 실행할 때 JUnit 플랫폼을 사용하겠다는 선언.
	useJUnitPlatform()
}
